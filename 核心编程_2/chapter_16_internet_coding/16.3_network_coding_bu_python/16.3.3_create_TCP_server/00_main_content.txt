ss = socket() # 创建服务器套接字
ss.bind() # 把地址绑定到套接字上
ss.listen() # 监听连接
inf_loop: # 服务器无限循环
    cs = ss.accept() # 接收客户端连接
"""
一个简单的（单线程的）服务器会调用accept()函数等待连接的到来
默认情况下，accept()函数是阻塞式的
一旦接收一个连接,accept()函数就会返回一个单独的客户端套接字用于后续的通信。
使用新的客户端套接字就想把客户的电话转给一个客户服务人员,空出主机等待下一个客户。
一个socket服务端，可以接受多个客户的请求
"""
comm_loop: # 通信循环
    cs.recv()/cs.send() # 对话（接收与发送）
cs.close() # 关闭客户端套接字
ss.close() # 关闭服务器套接字
